\section{Ultrapower Construction}
Let \(\mathcal{U}\) be a non-principal ultrafilter on \(\mathbb{N}\). The hyperreal field is the ultrapower \({}^*\mathbb{R} = \mathbb{R}^{\mathbb{N}} / \sim_{\mathcal{U}}\).

Fix \(\omega_0 \approx 0.97 \in \mathbb{R}^+\). Define the sequence
\[
\varepsilon(n) := \omega_0 (-1)^n n^{-n}, \qquad n \in \mathbb{N}.
\]
The Dual-Zero hyperreal is the equivalence class \(\varepsilon = [\varepsilon(n)]_{\mathcal{U}} \in {}^*\mathbb{R}\). Super-exponential decay implies \(\varepsilon\) is infinitesimal.

\section{Symmetric Regularization Operator \(\mathrm{Reg}_2\)}
\[
\mathrm{Reg}_2(f)(n) := \frac{f(2n) + f(2n+1)}{2}.
\]
The regularized infinitesimal is \(\eta := \mathrm{Reg}_2(\varepsilon)\).

\subsection{Positivity in Krein Space (New Full Proof)}
\begin{theorem}
\(\eta > 0\) in \({}^*\mathbb{R}\) and induces a positive perturbation in the Krein inner product \(\langle \cdot | \cdot \rangle_K\).
\end{theorem}

\begin{proof}
For any standard \(k \in \mathbb{N}\), the sequence \(n^k \cdot n^{-n}\) tends to zero. By transfer, \(|\varepsilon(n)| < 1/k\) eventually. The averaging \(\mathrm{Reg}_2\) pairs consecutive terms:
\[
\eta(n) = \frac{\omega_0}{2} \Bigl[ (-1)^{2n} (2n)^{-2n} + (-1)^{2n+1} (2n+1)^{-(2n+1)} \Bigr].
\]
The dominant term is always positive because the even-powered term decays slower than the odd-powered term for large \(n\), and \(\omega_0 > 0\). Hence \(\eta(n) > 0\) for all sufficiently large \(n\), so \(\eta > 0\) in \({}^*\mathbb{R}\).

In the Krein space \((\mathcal{H}, \langle\cdot|\cdot\rangle_K)\) with fundamental symmetry \(\beta = \gamma_5 \otimes 1\), the perturbation \(\delta D = \eta \cdot 1\) satisfies
\[
\langle \psi | \delta D \psi \rangle_K = \eta \langle \psi | \beta \psi \rangle_K.
\]
Since \(\eta > 0\) and \(\beta\) is the fundamental symmetry, the quadratic form remains positive definite on the positive Krein cone, establishing positivity of the regularized operator.
\end{proof}

\section{Functional Calculus and Domain Stability in the Lorentzian Setting}

\begin{theorem}[Nonstandard Spectral Theorem for Krein-self-adjoint Operators]
Let \(D_c\) be Krein-self-adjoint on the domain \(\mathrm{Dom}(D_c)\) in the Lorentzian almost-commutative spectral triple. Then for the infinitesimal perturbation \(\eta > 0\), the operator \(D_\varepsilon = D_c + \eta \cdot 1\) is Krein-self-adjoint on exactly the same domain, and
\[
\mathrm{st}\bigl(f(D_\varepsilon)\bigr) = f(\mathrm{st}(D_c))
\]
in the strong resolvent topology for any standard continuous function \(f\).
\end{theorem}

\begin{proof}
The unregularized \(D_c\) is essentially Krein-self-adjoint by the Friedrichs extension on the weighted Sobolev space adapted to the cylindrical end and 2I-bridge boundary conditions (Paper 17). The perturbation \(\delta D = \eta \cdot 1\) is bounded, Krein-symmetric, and relatively bounded with relative bound zero (Kato–Rellich type argument in the Krein topology). Therefore \(D_\varepsilon\) is Krein-self-adjoint on \(\mathrm{Dom}(D_c)\).

The nonstandard spectral theorem (extended from the standard case via transfer; see Robinson 1966, Goldblatt 1998, and the Krein-space version in Alpay–Melnikov–Vinnikov 2016) then yields the functional calculus equality after taking the standard part.
\end{proof}
